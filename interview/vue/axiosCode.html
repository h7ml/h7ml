二、实现一个简易版 axios#

构建一个Axios构造函数，核心代码为request

1class Axios {
2  constructor() {}
3
4  request(config) {
5    return new Promise((resolve) => {
6      const { url = '', method = 'get', data = {} } = config
7      // 发送ajax请求
8      const xhr = new XMLHttpRequest()
9      xhr.open(method, url, true)
10      xhr.onload = function () {
11        console.log(xhr.responseText)
12        resolve(xhr.responseText)
13      }
14      xhr.send(data)
15    })
16  }
17}

导出axios实例

1// 最终导出axios的方法，即实例的request方法
2function CreateAxiosFn() {
3  const axios = new Axios()
4  const req = axios.request.bind(axios)
5  return req
6}
7
8// 得到最后的全局变量axios
9const axios = CreateAxiosFn()

上述就已经能够实现axios({ })这种方式的请求

下面是来实现下axios.method()这种形式的请求

1// 定义get,post...方法，挂在到Axios原型上
2const methodsArr = ['get', 'delete', 'head', 'options', 'put', 'patch', 'post']
3methodsArr.forEach((met) => {
4  Axios.prototype[met] = function () {
5    console.log(`执行${met}方法`)
6    // 处理单个方法
7    if (['get', 'delete', 'head', 'options'].includes(met)) {
8      // 2个参数(url[, config])
9      return this.request({
10        method: met,
11        url: arguments[0],
12        ...(arguments[1] || {}),
13      })
14    }
15    else {
16      // 3个参数(url[,data[,config]])
17      return this.request({
18        method: met,
19        url: arguments[0],
20        data: arguments[1] || {},
21        ...(arguments[2] || {}),
22      })
23    }
24  }
25})

1const utils = {

10    }
11  },
12}

1request(config) {
2    // 拦截器和请求组装队列
3    let chain = [this.sendAjax.bind(this), undefined] // 成对出现的，失败回调暂时不处理
4
5    // 请求拦截
6    this.interceptors.request.handlers.forEach(interceptor => {
7        chain.unshift(interceptor.fullfield, interceptor.rejected)
8    })
9

11    this.interceptors.response.handlers.forEach(interceptor => {
12        chain.push(interceptor.fullfield, interceptor.rejected)
13    })
14

17    while(chain.length > 0) {
18        promise = promise.then(chain.shift(), chain.shift())
19    }
20    return promise;

chains大概是['fulfilled1','reject1','fulfilled2','reject2','this.sendAjax','undefined','fulfilled2','reject2','fulfilled1','reject1']这种形式

1Axios.prototype.request = function request(config) {
2  /*
3    先是 mergeConfig ... 等，不再阐述
4  */
5  // Hook up interceptors middleware 创建拦截器链. dispatchRequest 是重中之重，后续重点
6  const chain = [dispatchRequest, undefined]
7
8  // push各个拦截器方法 注意：interceptor.fulfilled 或 interceptor.rejected 是可能为undefined
9  this.interceptors.request.forEach((interceptor) => {
10    // 请求拦截器逆序 注意此处的 forEach 是自定义的拦截器的forEach方法
11    chain.unshift(interceptor.fulfilled, interceptor.rejected)
12  })
13
14  this.interceptors.response.forEach((interceptor) => {
15    // 响应拦截器顺序 注意此处的 forEach 是自定义的拦截器的forEach方法
16    chain.push(interceptor.fulfilled, interceptor.rejected)
17  })
18
19  // 初始化一个promise对象，状态为resolved，接收到的参数为已经处理合并过的config对象
20  let promise = Promise.resolve(config)
21
22  // 循环拦截器的链
23  while (chain.length)
24    promise = promise.then(chain.shift(), chain.shift()) // 每一次向外弹出拦截器
25
26  // 返回 promise
27  return promise
28}

InterceptorManager构造函数

1// 拦截器的初始化 其实就是一组钩子函数
2function InterceptorManager() {
3  this.handlers = []
4}
5
6// 调用拦截器实例的use时就是往钩子函数中push方法
7InterceptorManager.prototype.use = function use(fulfilled, rejected) {
8  this.handlers.push({
9    fulfilled,
10    rejected,
11  })
12  return this.handlers.length - 1
13}
14
15// 拦截器是可以取消的，根据use的时候返回的ID，把某一个拦截器方法置为null

17InterceptorManager.prototype.eject = function eject(id) {

19    this.handlers[id] = null
20}

1const isCancel = require('../cancel/isCancel')
2const defaults = require('../defaults')
3const utils = require('./../utils')

7
8// 判断请求是否已被取消，如果已经被取消，抛出已取消
9function throwIfCancellationRequested(config) {
10  if (config.cancelToken)
11    config.cancelToken.throwIfRequested()
12}
13
14module.exports = function dispatchRequest(config) {
15  throwIfCancellationRequested(config)
16
17  // 如果包含baseUrl, 并且不是config.url绝对路径，组合baseUrl以及config.url
18  if (config.baseURL && !isAbsoluteURL(config.url)) {
19    // 组合baseURL与url形成完整的请求路径
20    config.url = combineURLs(config.baseURL, config.url)
21  }
22
23  config.headers = config.headers || {}
24
25  // 使用/lib/defaults.js中的transformRequest方法，对config.headers和config.data进行格式化
26  // 比如将headers中的Accept，Content-Type统一处理成大写
27  // 比如如果请求正文是一个Object会格式化为JSON字符串，并添加application/json;charset=utf-8的Content-Type
28  // 等一系列操作
29  config.data = transformData(config.data, config.headers, config.transformRequest)
30
31  // 合并不同配置的headers，config.headers的配置优先级更高
32  config.headers = utils.merge(config.headers.common || {}, config.headers[config.method] || {}, config.headers || {})
33
34  // 删除headers中的method属性
35  utils.forEach(['delete', 'get', 'head', 'post', 'put', 'patch', 'common'], (method) => {
36    delete config.headers[method]
37  })
38
39  // 如果config配置了adapter，使用config中配置adapter的替代默认的请求方法
40  const adapter = config.adapter || defaults.adapter
41
42  // 使用adapter方法发起请求（adapter根据浏览器环境或者Node环境会有不同）
43  return adapter(config).then(
44    // 请求正确返回的回调
45    (response) => {
46      // 判断是否以及取消了请求，如果取消了请求抛出以取消
47      throwIfCancellationRequested(config)
48
49      // 使用/lib/defaults.js中的transformResponse方法，对服务器返回的数据进行格式化
50      // 例如，使用JSON.parse对响应正文进行解析
51      response.data = transformData(response.data, response.headers, config.transformResponse)
52
53      return response
54    },
55    // 请求失败的回调
56    (reason) => {
57      if (!isCancel(reason)) {
58        throwIfCancellationRequested(config)
59
60        if (reason && reason.response)
61          reason.response.data = transformData(reason.response.data, reason.response.headers, config.transformResponse)
62      }
63      return Promise.reject(reason)
64    }
65  )
66}

再来看看axios是如何实现取消请求的，实现文件在CancelToken.js

1function CancelToken(executor) {
2  if (typeof executor !== 'function')
3    throw new TypeError('executor must be a function.')
4
5  // 在 CancelToken 上定义一个 pending 状态的 promise ，将 resolve 回调赋值给外部变量 resolvePromise
6  let resolvePromise
7  this.promise = new Promise((resolve) => {
8    resolvePromise = resolve
9  })
10
11  const token = this
12  // 立即执行 传入的 executor函数，将真实的 cancel 方法通过参数传递出去。
13  // 一旦调用就执行 resolvePromise 即前面的 promise 的 resolve，就更改promise的状态为 resolve。
14  // 那么xhr中定义的 CancelToken.promise.then方法就会执行, 从而xhr内部会取消请求
15  executor((message) => {
16    // 判断请求是否已经取消过，避免多次执行
17    if (token.reason)
18      return
19
20    token.reason = new Cancel(message)
21    resolvePromise(token.reason)
22  })
23}
24
25CancelToken.source = function source() {
26  // source 方法就是返回了一个 CancelToken 实例，与直接使用 new CancelToken 是一样的操作
27  let cancel
28  const token = new CancelToken((c) => {
29    cancel = c
30  })
31  // 返回创建的 CancelToken 实例以及取消方法
32  return {
33    token,
34    cancel,
35  }
36}
